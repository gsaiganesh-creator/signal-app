import pandas as pd, numpy as np
RSI_LOW, RSI_HIGH = 35, 70
EMA_FAST, EMA_SLOW = 20, 50
STOP_LOSS, TARGET = 2.5, 6.0
def signal(df):
r = rsi(df["close"])
fast = df["close"].ewm(span=EMA_FAST).mean()
slow = df["close"].ewm(span=EMA_SLOW).mean()
if r.iloc[-1] < RSI_LOW and fast > slow:
return "BUY"
if r.iloc[-1] > RSI_HIGH:
return "SELL"
return "HOLD"
