Real-time quote — Node.js backend

// packages/api/src/mstock.js
const axios = require('axios');

const MSTOCK_BASE = 'https://api.mstock.com/v1';

async function getQuote(symbol) {
  const res = await axios.get(`${MSTOCK_BASE}/quote/${symbol}`, {
    headers: {
      'x-api-key': process.env.MSTOCK_API_KEY,
      'Authorization': `Bearer ${process.env.MSTOCK_TOKEN}`
    }
  });
  return res.data;
}

const kiteLoginUrl = `https://kite.zerodha.com/connect/login?v=3&api_key=${KITE_API_KEY}`;

// Step 2: User logs in, Kite redirects to your callback with ?request_token=xxx
app.get('/auth/kite/callback', async (req, res) => {
  const { request_token } = req.query;
  // Exchange request_token for access_token
  const session = await kite.generateSession(request_token, KITE_API_SECRET);
  // Store encrypted access_token in Supabase against user.id
  await supabase.from('broker_tokens').upsert({
    user_id: req.user.id,
    broker: 'zerodha',
    access_token: encrypt(session.access_token),
    expires_at: new Date(Date.now() + 86400000)
  });
  res.redirect('/dashboard');
});

// Step 3: Fetch live portfolio holdings
async function getPortfolio(userId) {
  const { data } = await supabase
    .from('broker_tokens').select().eq('user_id', userId).single();
  const kite = new KiteConnect({ api_key: KITE_API_KEY });
  kite.setAccessToken(decrypt(data.access_token));
  return kite.getHoldings(); // Live holdings with LTP, P&L
}

6 · ML Model: Random Forest Signal Generator

# packages/ml/signal_model.py
import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.preprocessing import StandardScaler
from ta import add_all_ta_features  # pip install ta

def engineer_features(df):
    df = add_all_ta_features(df, open="open", high="high", low="low",
                              close="close", volume="volume")
    df['target'] = (df['close'].shift(-3) > df['close'] * 1.02).astype(int)
    # Target = 1 if stock rises >2% in next 3 days
    return df.dropna()

FEATURES = ['momentum_rsi', 'trend_macd', 'volatility_bbh',
            'volume_obv', 'trend_adx', 'momentum_stoch_signal']

model = RandomForestClassifier(n_estimators=200, max_depth=8,
                                min_samples_leaf=5, random_state=42)

def train(df):
    df = engineer_features(df)
    X = df[FEATURES]; y = df['target']
    model.fit(X[:-60], y[:-60])  # hold last 60 days for validation

def predict(df):
    df = engineer_features(df)
    proba = model.predict_proba(df[FEATURES].tail(1))[0][1]
    return {
      "signal": "BUY" if proba > 0.65 else "HOLD",
      "confidence": round(proba * 100, 1),
      "rf_score": round(proba, 3)
    }

// Abstract data provider pattern — swap mStock ↔ Kite ↔ Upstox
class DataProvider {
  constructor(provider = process.env.DATA_PROVIDER) {
    if (provider === 'kite')    this.client = new KiteProvider();
    if (provider === 'mstock') this.client = new MStockProvider();
    if (provider === 'upstox') this.client = new UpstoxProvider();
  }
  getQuote(symbol)             { return this.client.getQuote(symbol); }
  subscribe(symbols, callback) { return this.client.subscribe(symbols, callback); }
  getHistory(symbol, from, to) { return this.client.getHistory(symbol, from, to); }
}

// Switch: DATA_PROVIDER=kite → done. Zero other code changes.
